   1	/// Context Offloading - Save verbose tool outputs to external files
   2	/// 
   3	/// Inspired by TencentDB-Agent-Memory's approach:
   4	/// - Complete results saved to .ox/refs/{task_id}_{step}.md
   5	/// - Message context only keeps summary + node_id reference
   6	/// - Reduces token usage by 60%+ for long-running tasks
   7	/// - Maintains full traceability via node_id lookup
   8	
   9	use std::path::{Path, PathBuf};
  10	use std::fs;
  11	use chrono::Utc;
  12	
  13	/// Represents a tool execution result with offloading support
  14	#[derive(Debug, Clone)]
  15	pub struct OffloadedResult {
  16	    /// Summary of the result (kept in context)
  17	    pub summary: String,
  18	    /// Reference ID for retrieving full content (node_id)
  19	    pub node_id: String,
  20	    /// Path to the full content file (if offloaded)
  21	    pub ref_path: Option<PathBuf>,
  22	    /// Whether this was actually offloaded
  23	    pub is_offloaded: bool,
  24	}
  25	
  26	impl OffloadedResult {
  27	    /// Create a new offloaded result
  28	    pub fn new(summary: String, node_id: String, ref_path: Option<PathBuf>) -> Self {
  29	        let is_offloaded = ref_path.is_some();
  30	        Self {
  31	            summary,
  32	            node_id,
  33	            ref_path,
  34	            is_offloaded,
  35	        }
  36	    }
  37	
  38	    /// Format as message content for LLM context
  39	    pub fn to_context_message(&self) -> String {
  40	        if self.is_offloaded {
  41	            format!(
  42	                "✅ Result saved to: {}\n📄 Summary: {}\n💡 Use `recall {}` to view full details",
  43	                self.ref_path.as_ref().unwrap().display(),
  44	                self.summary,
  45	                self.node_id
  46	            )
  47	        } else {
  48	            self.summary.clone()
  49	        }
  50	    }
  51	}

  53	/// Context offloader - manages saving and retrieving tool outputs
  54	pub struct ContextOffloader {
  55	    /// Base directory for storing references (.ox/refs/)
  56	    refs_dir: PathBuf,
  57	    /// Current task/session ID for organizing refs
  58	    task_id: String,
  59	}
  60	
  61	impl ContextOffloader {
  62	    /// Create a new context offloader
  63	    pub fn new(working_dir: &Path, task_id: &str) -> Self {
  64	        let refs_dir = working_dir.join(".ox").join("refs");
  65	        
  66	        // Ensure refs directory exists
  67	        if let Err(e) = fs::create_dir_all(&refs_dir) {
  68	            tracing::warn!("Failed to create refs directory: {}", e);
  69	        }
  70	        
  71	        Self {
  72	            refs_dir,
  73	            task_id: task_id.to_string(),
  74	        }
  75	    }
  76	
  77	    /// Process a tool result and decide whether to offload
  78	    /// 
  79	    /// Offloading criteria:
  80	    /// - Content length > threshold (default: 2000 chars)
  81	    /// - Contains file paths or code blocks
  82	    /// - Is a search/listing operation result
  83	    pub fn process_result(
  84	        &self,
  85	        tool_name: &str,
  86	        content: &str,
  87	        step_index: usize,
  88	        threshold: usize,
  89	    ) -> OffloadedResult {
  90	        let should_offload = self.should_offload(tool_name, content, threshold);
  91	        
  92	        if should_offload {
  93	            self.offload_result(tool_name, content, step_index)
  94	        } else {
  95	            // Keep in context without offloading
  96	            OffloadedResult::new(
  97	                content.to_string(),
  98	                format!("inline_{}", step_index),
  99	                None,
 100	            )
 101	        }
 102	    }
 103	
 104	    /// Determine if a result should be offloaded
 105	    fn should_offload(&self, tool_name: &str, content: &str, threshold: usize) -> bool {
 106	        // Always offload if content exceeds threshold
 107	        if content.len() > threshold {
 108	            return true;
 109	        }
 110	
 111	        // Offload specific tool types that tend to produce verbose output
 112	        match tool_name {
 113	            "code_search" | "file_list" | "grep" => {
 114	                // Search results often contain many file paths
 115	                content.lines().count() > 10
 116	            }
 117	            "shell_exec" => {
 118	                // Shell output can be very verbose
 119	                content.len() > 1000 || content.lines().count() > 20
 120	            }
 121	            "file_read" => {
 122	                // Large file reads
 123	                content.len() > 1500
 124	            }
 125	            _ => false,
 126	        }
 127	    }
 128	
 129	    /// Offload a result to external file
 130	    fn offload_result(
 131	        &self,
 132	        tool_name: &str,
 133	        content: &str,
 134	        step_index: usize,
 135	    ) -> OffloadedResult {
 136	        let timestamp = Utc::now().format("%Y%m%d_%H%M%S");
 137	        let node_id = format!("{}_step{}_{}", self.task_id, step_index, timestamp);
 138	        let filename = format!("{}.md", node_id);
 139	        let ref_path = self.refs_dir.join(&filename);
 140	
 141	        // Generate summary (first 200 chars + line count)
 142	        let summary = if content.len() > 200 {
 143	            format!(
 144	                "{}... ({} lines total)",
 145	                content.get(..200).unwrap_or(content),
 146	                content.lines().count()
 147	            )
 148	        } else {
 149	            content.to_string()
 150	        };
 151	
 152	        // Write full content to file
 153	        let file_content = format!(
 154	            "# Tool Execution Result\n\n\
 155	             **Tool**: {}\n\
 156	             **Step**: {}\n\
 157	             **Timestamp**: {}\n\
 158	             **Node ID**: {}\n\n\
 159	             ---\n\n\
 160	             {}\n",
 161	            tool_name,
 162	            step_index,
 163	            Utc::now().to_rfc3339(),
 164	            node_id,
 165	            content
 166	        );
 167	
 168	        if let Err(e) = fs::write(&ref_path, &file_content) {
 169	            tracing::error!("Failed to write offloaded result to {}: {}", ref_path.display(), e);
 170	            // Fallback: keep in context
 171	            return OffloadedResult::new(
 172	                content.to_string(),
 173	                format!("fallback_{}", step_index),
 174	                None,
 175	            );
 176	        }
 177	
 178	        tracing::info!(
 179	            "Offloaded tool result: {} -> {} ({} bytes)",
 180	            tool_name,
 181	            ref_path.display(),
 182	            file_content.len()
 183	        );
 184	
 185	        OffloadedResult::new(summary, node_id, Some(ref_path))
 186	    }
 187	
 188	    /// Retrieve full content from an offloaded reference
 189	    pub fn retrieve_full_content(&self, node_id: &str) -> Option<String> {
 190	        let filename = format!("{}.md", node_id);
 191	        let ref_path = self.refs_dir.join(&filename);
 192	
 193	        if ref_path.exists() {
 194	            match fs::read_to_string(&ref_path) {
 195	                Ok(content) => {
 196	                    tracing::info!("Retrieved full content for node_id: {}", node_id);
 197	                    Some(content)
 198	                }
 199	                Err(e) => {
 200	                    tracing::error!("Failed to read offloaded content {}: {}", ref_path.display(), e);
 201	                    None
 202	                }
 203	            }
 204	        } else {
 205	            tracing::warn!("Offloaded content not found for node_id: {}", node_id);
 206	            None
 207	        }
 208	    }

 210	    /// Clean up old references (keep last N per task)
 211	    pub fn cleanup_old_refs(&self, keep_count: usize) -> Result<usize, String> {
 212	        if !self.refs_dir.exists() {
 213	            return Ok(0);
 214	        }
 215	
 216	        let mut entries: Vec<_> = fs::read_dir(&self.refs_dir)
 217	            .map_err(|e| format!("Failed to read refs dir: {}", e))?
 218	            .filter_map(|entry| entry.ok())
 219	            .filter(|entry| {
 220	                entry.file_name()
 221	                    .to_string_lossy()
 222	                    .starts_with(&self.task_id)
 223	            })
 224	            .collect();
 225	
 226	        // Sort by modification time (oldest first)
 227	        entries.sort_by_key(|entry| {
 228	            entry.metadata()
 229	                .ok()
 230	                .and_then(|m| m.modified().ok())
 231	                .unwrap_or_else(|| std::time::SystemTime::UNIX_EPOCH)
 232	        });
 233	
 234	        let total_count = entries.len();
 235	        let to_delete = total_count.saturating_sub(keep_count);
 236	
 237	        let mut deleted = 0;
 238	        for entry in entries.iter().take(to_delete) {
 239	            if let Err(e) = fs::remove_file(entry.path()) {
 240	                tracing::warn!("Failed to delete old ref {}: {}", entry.path().display(), e);
 241	            } else {
 242	                deleted += 1;
 243	            }
 244	        }
 245	
 246	        if deleted > 0 {
 247	            tracing::info!("Cleaned up {} old references (kept {})", deleted, keep_count);
 248	        }
 249	
 250	        Ok(deleted)
 251	    }
 252	}

 260	    fn test_offload_large_content() {
