   5	/// Summarize tool result for display.
   6	pub fn summarize_tool_result(name: &str, output: &str) -> String {
   7	    match name {
   8	        "file_write" | "file_patch" => {
   9	            let first_line = output.lines().next().unwrap_or(output);
  10	            let truncated: String = first_line.chars().take(120).collect();
  11	            if first_line.len() > 120 {
  12	                format!("{truncated}...")
  13	            } else {
  14	                truncated
  15	            }
  16	        }
  17	        "file_read" => {
  18	            let line_count = output.lines().count();
  19	            let first_path = output
  20	                .lines()
  21	                .next()
  22	                .and_then(|l| l.split_whitespace().next())
  23	                .unwrap_or("");
  24	            if first_path.is_empty() {
  25	                format!("{line_count} lines")
  26	            } else {
  27	                format!("{first_path} ({line_count} lines)")
  28	            }
  29	        }
  30	        "code_search" => {
  31	            let match_count = output.lines().take(101).count();
  32	            if output.contains("truncated") {
  33	                format!("100+ matches")
  34	            } else if match_count == 0 {
  35	                "no matches".into()
  36	            } else {
  37	                format!("{match_count} matches")
  38	            }
  39	        }
  40	        "shell_exec" => {
  41	            if let Some(line) = output.lines().find(|l| l.starts_with("[exit code:")) {
  42	                format!("{line}")
  43	            } else {
  44	                let count = output.lines().count();
  45	                format!("{count} lines")
  46	            }
  47	        }
  48	        "file_list" | "file_search" => {
  49	            let count = output.lines().count();
  50	            format!("{count} entries")
  51	        }
  52	        "project_detect" => {
  53	            let first_line = output.lines().next().unwrap_or(output);
  54	            let truncated: String = first_line.chars().take(120).collect();
  55	            truncated
  56	        }
  57	        "git_status" | "git_diff" | "git_commit" => {
  58	            let count = output.lines().count();
  59	            format!("{count} lines")
  60	        }
  61	        "web_fetch" => {
  62	            let len = output.len();
  63	            format!("{len} chars")
  64	        }
  65	        _ => {
  66	            let truncated: String = output.chars().take(120).collect();
  67	            if output.len() > 120 {
  68	                format!("{truncated}...")
  69	            } else {
  70	                truncated
  71	            }
  72	        }
  73	    }
  74	}

  77	pub fn extract_file_path_from_output(output: &str) -> Option<String> {
  78	    if let Some(pos) = output.find("to ") {
  79	        // 使用安全的字符边界检查
  80	        if let Some(after_to) = output.get(pos + 3..) {
  81	            if let Some(end_pos) = after_to.find('\n') {
  82	                // 安全地获取子字符串
  83	                after_to.get(..end_pos).map(|s| s.trim().to_string())
  84	            } else {
  85	                Some(after_to.trim().to_string())
  86	            }
  87	        } else {
  88	            None
  89	        }
  90	    } else {
  91	        None
  92	    }

 114	/// Calculate tool success rate from session messages.
 115	pub fn calculate_tool_success_rate(messages: &[Message]) -> f64 {
 116	    let mut total_tools = 0u32;
 117	    let mut successful_tools = 0u32;
 118	
 119	    for msg in messages {
 120	        if let Message::ToolResult { content, .. } = msg {
 121	            total_tools += 1;
 122	            if !content.starts_with("Error:") && !content.starts_with("Unknown tool") {
 123	                successful_tools += 1;
 124	            }
 125	        }
 126	    }
 127	
 128	    if total_tools == 0 {
 129	        1.0
 130	    } else {
 131	        successful_tools as f64 / total_tools as f64
 132	    }
 133	}
