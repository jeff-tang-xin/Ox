  39	    pub fn parse_line(&mut self, line: &str) -> Vec<LlmStreamEvent> {
  40	        let line = line.trim_end_matches('\r');
  41	        if self.buffer.push_line(line) {
  42	            let data = self.buffer.take_data();
  43	            self.process_raw_data(&data)
  44	        } else {
  45	            Vec::new()
  46	        }
  47	    }

  49	    pub fn parse_chunk(&mut self, chunk: &str) -> Vec<LlmStreamEvent> {
  50	        let mut events = Vec::new();
  51	        for line in chunk.lines() {
  52	            events.extend(self.parse_line(line));
  53	        }
  54	        events
  55	    }
  56	
  57	    fn process_raw_data(&mut self, data: &str) -> Vec<LlmStreamEvent> {
  58	        let mut events = Vec::new();
  59	        let data = data.trim();
  60	        if data.is_empty() || data == "[DONE]" {
  61	            return events;
  62	        }
  63	        for result in parse_json_values(data) {
  64	            match result {
  65	                Ok(json) => events.extend(self.process_json_value(&json)),
  66	                Err(e) => tracing::warn!("Failed to parse SSE data: {}", e),
  67	            }
  68	        }
  69	        events
  70	    }
  71	
  72	    fn process_json_value(&mut self, json: &serde_json::Value) -> Vec<LlmStreamEvent> {
  73	        let mut events = Vec::new();
  74	        if let Some(choices) = json.get("choices").and_then(|c| c.as_array()) {
  75	            for choice in choices {
  76	                let index = choice.get("index").and_then(|i| i.as_u64()).unwrap_or(0);
  77	                self.seen_indices.insert(index);
  78	
  79	                if let Some(finish_reason) = choice.get("finish_reason").and_then(|f| f.as_str()) {
  80	                    if finish_reason == "tool_calls" || finish_reason == "stop" {
  81	                        if self.active_tool_calls.contains(&index) {
  82	                            if let Some(id) = self.tool_call_ids.get(&index) {
  83	                                events.push(LlmStreamEvent::ToolCallEnd { id: id.clone() });
  84	                            }
  85	                            self.active_tool_calls.remove(&index);
  86	                        }
  87	                    }
  88	                }
  89	
  90	                if let Some(delta) = choice.get("delta") {
  91	                    events.extend(self.process_delta(delta, index));
  92	                }
  93	            }
  94	        }
  95	
  96	        if let Some(usage) = json.get("usage").filter(|u| !u.is_null()) {
  97	            events.push(LlmStreamEvent::Done {
  98	                usage: TokenUsage {
  99	                    prompt_tokens: usage
 100	                        .get("prompt_tokens")
 101	                        .and_then(|v| v.as_u64())
 102	                        .unwrap_or(0) as u32,
 103	                    completion_tokens: usage
 104	                        .get("completion_tokens")
 105	                        .and_then(|v| v.as_u64())
 106	                        .unwrap_or(0) as u32,
 107	                    total_tokens: usage
 108	                        .get("total_tokens")
 109	                        .and_then(|v| v.as_u64())
 110	                        .unwrap_or(0) as u32,
 111	                },
 112	            });
 113	        }
 114	        events
 115	    }

 117	    fn process_delta(&mut self, delta: &serde_json::Value, index: u64) -> Vec<LlmStreamEvent> {
 118	        let mut events = Vec::new();
 119	
 120	        if let Some(content) = delta
 121	            .get("content")
 122	            .and_then(|c| c.as_str())
 123	            .filter(|s| !s.is_empty())
 124	        {
 125	            events.push(LlmStreamEvent::TextDelta(content.to_string()));
 126	        }
 127	
 128	        if let Some(tool_calls) = delta.get("tool_calls").and_then(|t| t.as_array()) {
 129	            for tc in tool_calls {
 130	                let tc_index = tc.get("index").and_then(|i| i.as_u64()).unwrap_or(index);
 131	
 132	                let id = tc
 133	                    .get("id")
 134	                    .and_then(|i| i.as_str())
 135	                    .map(|s| s.to_string())
 136	                    .or_else(|| {
 137	                        tc.get("function")
 138	                            .and_then(|f| f.get("id"))
 139	                            .and_then(|i| i.as_str())
 140	                            .map(|s| s.to_string())
 141	                    });
 142	
 143	                let name = tc
 144	                    .get("name")
 145	                    .and_then(|n| n.as_str())
 146	                    .filter(|s| !s.is_empty())
 147	                    .map(|s| s.to_string())
 148	                    .or_else(|| {
 149	                        tc.get("function")
 150	                            .and_then(|f| f.get("name"))
 151	                            .and_then(|n| n.as_str())
 152	                            .filter(|s| !s.is_empty())
 153	                            .map(|s| s.to_string())
 154	                    });
 155	
 156	                if let (Some(id), Some(name)) = (&id, &name) {
 157	                    // O(1) lookup using reverse map
 158	                    let is_new = !self.id_to_index.contains_key(id.as_str());
 159	                    if is_new {
 160	                        self.tool_call_ids.insert(tc_index, id.clone());
 161	                        self.tool_call_names.insert(tc_index, name.clone());
 162	                        self.active_tool_calls.insert(tc_index);
 163	                        self.id_to_index.insert(id.clone(), tc_index);
 164	                        events.push(LlmStreamEvent::ToolCallStart {
 165	                            id: id.clone(),
 166	                            name: name.clone(),
 167	                        });
 168	                    }
 169	                }
 170	
 171	                let args = tc
 172	                    .get("arguments")
 173	                    .and_then(|a| a.as_str())
 174	                    .map(|s| s.to_string())
 175	                    .or_else(|| {
 176	                        tc.get("function")
 177	                            .and_then(|f| f.get("arguments"))
 178	                            .and_then(|a| a.as_str())
 179	                            .map(|s| s.to_string())
 180	                    });
 181	
 182	                if let Some(args) = args {
 183	                    let buffer = self
 184	                        .argument_buffers
 185	                        .entry(tc_index)
 186	                        .or_insert_with(String::new);
 187	                    buffer.push_str(&args);
 188	                    let tc_id = self
 189	                        .tool_call_ids
 190	                        .get(&tc_index)
 191	                        .cloned()
 192	                        .unwrap_or_else(|| format!("tool_call_{}", tc_index));
 193	                    events.push(LlmStreamEvent::ToolCallArgumentsDelta {
 194	                        id: tc_id,
 195	                        delta: args,
 196	                    });
 197	                }
 198	            }
 199	        }
 200	        events
 201	    }

 204	        let mut events = Vec::new();
 205	        for index in self.active_tool_calls {

 208	            }
