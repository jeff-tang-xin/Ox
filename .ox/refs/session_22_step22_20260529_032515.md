 214	    /// 基于规则分类
 215	    pub fn classify(&self, user_message: &str) -> IntentInfo {
 216	        let msg_lower = user_message.to_lowercase();
 217	        let words = self.extract_words(&msg_lower);
 218	
 219	        // 统计各类型得分
 220	        let mut scores = std::collections::HashMap::new();
 221	        scores.insert(QuestionType::CodeReading, 0);
 222	        scores.insert(QuestionType::CodeWriting, 0);
 223	        scores.insert(QuestionType::Debugging, 0);
 224	        scores.insert(QuestionType::Refactoring, 0);
 225	        scores.insert(QuestionType::Exploration, 0);
 226	
 227	        for word in &words {
 228	            if self.action_verbs.read_verbs.contains(word.as_str()) {
 229	                *scores.get_mut(&QuestionType::CodeReading).unwrap() += 1;
 230	            }
 231	            if self.action_verbs.write_verbs.contains(word.as_str()) {
 232	                *scores.get_mut(&QuestionType::CodeWriting).unwrap() += 1;
 233	            }
 234	            if self.action_verbs.search_verbs.contains(word.as_str()) {
 235	                *scores.get_mut(&QuestionType::CodeReading).unwrap() += 1;
 236	            }
 237	            if self.action_verbs.debug_verbs.contains(word.as_str()) {
 238	                *scores.get_mut(&QuestionType::Debugging).unwrap() += 1;
 239	            }
 240	            if self.action_verbs.refactor_verbs.contains(word.as_str()) {
 241	                *scores.get_mut(&QuestionType::Refactoring).unwrap() += 1;
 242	            }
 243	            if self.action_verbs.explore_verbs.contains(word.as_str()) {
 244	                *scores.get_mut(&QuestionType::Exploration).unwrap() += 1;
 245	            }
 246	        }
 247	
 248	        // 找到最高分的意图
 249	        let (best_intent, best_score) = scores
 250	            .into_iter()
 251	            .max_by_key(|&(_, score)| score)
 252	            .unwrap_or((QuestionType::GeneralQuestion, 0));
 253	
 254	        // 计算置信度
 255	        let total_words = words.len().max(1);
 256	        let confidence = if best_score == 0 {
 257	            0.3
 258	        } else {
 259	            ((best_score as f32 / total_words as f32) * 2.0).min(0.9)
 260	        };
 261	
 262	        // 提取关键词（简单实现：取前 5 个有意义的词）
 263	        let keywords: Vec<String> = words
 264	            .into_iter()
 265	            .filter(|w| w.len() > 1)
 266	            .take(5)
 267	            .collect();
 268	
 269	        // 根据意图推荐工具
 270	        let suggested_tools = self.recommend_tools(&best_intent);
 271	        
 272	        // === 新增：记忆检索决策 ===
 273	        let (should_search_memory, memory_query, memory_scope) = 
 274	            self.decide_memory_search(user_message, &best_intent, &keywords);
 275	
 276	        IntentInfo {
 277	            intent: best_intent,
 278	            confidence,
 279	            keywords,
 280	            suggested_tools,
 281	            should_search_memory,
 282	            memory_query,
 283	            memory_scope,
 284	        }
 285	    }

 287	    /// 决策是否需要检索记忆
 288	    fn decide_memory_search(
 289	        &self,
 290	        user_message: &str,
 291	        intent: &QuestionType,
 292	        keywords: &[String],
 293	    ) -> (bool, Option<String>, MemoryScope) {
 294	        let msg_lower = user_message.to_lowercase();
 295	        
 296	        // 规则 1: 用户明确提到历史/之前的内容
 297	        let history_indicators = [
 298	            "之前", "以前", "上次", "记得", "历史", "before", "previous", 
 299	            "remember", "last time", "earlier"
 300	        ];
 301	        
 302	        for indicator in &history_indicators {
 303	            if msg_lower.contains(indicator) {
 304	                return (
 305	                    true,
 306	                    Some(self.extract_memory_query(user_message, keywords)),
 307	                    MemoryScope::Project, // 历史内容通常在项目级别
 308	                );
 309	            }
 310	        }
 311	        
 312	        // 规则 2: 涉及项目特定知识（架构、约定等）
 313	        let project_knowledge_indicators = [
 314	            "架构", "结构", "规范", "约定", "风格", "习惯",
 315	            "architecture", "convention", "pattern", "style"
 316	        ];
 317	        
 318	        for indicator in &project_knowledge_indicators {
 319	            if msg_lower.contains(indicator) {
 320	                return (
 321	                    true,
 322	                    Some(self.extract_memory_query(user_message, keywords)),
 323	                    MemoryScope::Project,
 324	                );
 325	            }
 326	        }
 327	        
 328	        // 规则 3: 复杂任务可能需要参考最佳实践
 329	        match intent {
 330	            QuestionType::CodeWriting | QuestionType::Refactoring => {
 331	                if keywords.len() >= 3 { // 任务较复杂
 332	                    return (
 333	                        true,
 334	                        Some(self.extract_memory_query(user_message, keywords)),
 335	                        MemoryScope::Both, // 可能用到全局知识
 336	                    );
 337	                }
 338	            }
 339	            QuestionType::Debugging => {
 340	                // Debug 时查找类似问题的解决方案
 341	                return (
 342	                    true,
 343	                    Some(self.extract_memory_query(user_message, keywords)),
 344	                    MemoryScope::Both,
 345	                );
 346	            }
 347	            _ => {}
 348	        }
 349	        
 350	        // 默认不检索
 351	        (false, None, MemoryScope::Both)
 352	    }

 354	    /// 从用户消息中提取记忆检索查询词
 355	    fn extract_memory_query(&self, user_message: &str, keywords: &[String]) -> String {
 356	        // 简单策略：使用关键词作为查询词
 357	        if !keywords.is_empty() {
 358	            keywords.join(" ")
 359	        } else {
 360	            // 如果没有关键词，使用原始消息的前 50 个字符
 361	            user_message.chars().take(50).collect()
 362	        }
 363	    }

 365	    /// 提取单词（支持多语言）
 366	    fn extract_words(&self, text: &str) -> Vec<String> {
 367	        let mut words = Vec::new();
 368	
 369	        // 1. 英文单词：按空格和标点分割
 370	        for word in text.split_whitespace() {
 371	            let cleaned = word.trim_matches(|c: char| !c.is_alphanumeric());
 372	            if !cleaned.is_empty() {
 373	                words.push(cleaned.to_string());
 374	            }
 375	        }
 376	
 377	        // 2. 中文字符：每个字符单独作为一个"词"
 378	        for ch in text.chars() {
 379	            if ch.is_ascii() {
 380	                continue;
 381	            }
 382	            words.push(ch.to_string());
 383	        }
 384	
 385	        words
 386	    }

 451	/// 从 LLM 响应中提取意图
 452	pub fn extract_intent_from_llm_response(response: &str) -> Option<IntentInfo> {
 453	    // 查找最后一个 JSON 代码块
 454	    if let Some(json_start) = response.rfind("```json") {
 455	        if let Some(json_end) = response[json_start..].find("```") {
 456	            // 使用字符边界安全的切片方法
 457	            let start_byte = json_start + 7; // "```json" 的长度是7
 458	            let end_byte = json_start + json_end;
 459	            
 460	            // 确保字节索引在有效范围内且位于字符边界
 461	            if start_byte <= end_byte && end_byte <= response.len() {
 462	                // 使用 get() 方法进行安全切片
 463	                if let Some(json_str) = response.get(start_byte..end_byte) {
 464	                    match serde_json::from_str::<IntentInfo>(json_str.trim()) {
 465	                        Ok(info) => {
 466	                            tracing::info!(
 467	                                "Extracted intent from LLM: {:?} (confidence: {:.2})",
 468	                                info.intent,
 469	                                info.confidence
 470	                            );
 471	                            return Some(info);
 472	                        }
 473	                        Err(e) => {
 474	                            tracing::warn!("Failed to parse intent JSON: {}", e);
 475	                        }
 476	                    }
 477	                } else {
 478	                    tracing::warn!("Invalid UTF-8 boundaries in response");
 479	                }
 480	            }
 481	        }
 482	    }
 483	
 484	    None
 485	}

 487	/// 从响应中移除意图 JSON 块
 488	pub fn remove_intent_json(response: &str) -> String {
 489	    if let Some(json_start) = response.rfind("```json") {
 490	        if let Some(json_end) = response[json_start..].find("```") {
 491	            let json_block_end = json_start + json_end + 3;
 492	
 493	            // 确保字节索引在有效范围内
 494	            if json_start <= response.len() && json_block_end <= response.len() {
 495	                // 使用 get() 方法进行安全切片
 496	                if let (Some(before_part), Some(after_part)) = (
 497	                    response.get(..json_start),
 498	                    response.get(json_block_end..)
 499	                ) {
 500	                    // 移除 JSON 块及其前后的空行
 501	                    let before = before_part.trim_end();
 502	                    let after = after_part.trim_start();
 503	
 504	                    format!("{}\n{}", before, after)
 505	                } else {
 506	                    response.to_string()
 507	                }
 508	            } else {
 509	                response.to_string()
 510	            }
 511	        } else {
 512	            response.to_string()
 513	        }
 514	    } else {
 515	        response.to_string()
 516	    }
 517	}
