 194	    /// Load Skills from filesystem and register them
 195	    pub fn load_skills(&mut self, rt_env: &crate::runtime::RuntimeEnvironment) -> anyhow::Result<()> {

 219	    /// Get all tool schemas for LLM API calls (includes Skills as special tools).
 220	    pub fn schemas(&self) -> Vec<crate::llm::ToolSchema> {
 221	        let mut schemas: Vec<crate::llm::ToolSchema> = self.tools
 222	            .values()
 223	            .map(|t| crate::llm::ToolSchema {
 224	                name: t.name().to_string(),
 225	                description: t.description().to_string(),
 226	                parameters: t.parameters_schema(),
 227	            })
 228	            .collect();
 229	        
 230	        // Add Skills as special composite tools
 231	        for skill in &self.skills {
 232	            schemas.push(crate::llm::ToolSchema {
 233	                name: format!("skill:{}", skill.id),
 234	                description: format!(
 235	                    "Skill: {} - {}\n\n{}",
 236	                    skill.name,
 237	                    skill.description,
 238	                    skill.content
 239	                ),
 240	                parameters: serde_json::json!({
 241	                    "type": "object",
 242	                    "properties": {},
 243	                    "description": "This skill provides guidance. No parameters needed."
 244	                }),
 245	            });
 246	        }
 247	        
 248	        schemas
 249	    }
