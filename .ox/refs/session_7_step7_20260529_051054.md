 104	    /// Determine if a result should be offloaded
 105	    fn should_offload(&self, tool_name: &str, content: &str, threshold: usize) -> bool {
 106	        // Always offload if content exceeds threshold
 107	        if content.len() > threshold {
 108	            return true;
 109	        }
 110	
 111	        // Offload specific tool types that tend to produce verbose output
 112	        match tool_name {
 113	            "code_search" | "file_list" | "grep" => {
 114	                // Search results often contain many file paths
 115	                content.lines().count() > 10
 116	            }
 117	            "shell_exec" => {
 118	                // Shell output can be very verbose
 119	                content.len() > 1000 || content.lines().count() > 20
 120	            }
 121	            "file_read" => {
 122	                // Large file reads
 123	                content.len() > 1500
 124	            }
 125	            _ => false,
 126	        }
 127	    }
