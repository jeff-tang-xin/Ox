 156	             **Step**: {}\n\
 157	             **Timestamp**: {}\n\
 158	             **Node ID**: {}\n\n\

 160	             {}\n",

 163	            Utc::now().to_rfc3339(),
 164	            node_id,

 168	        if let Err(e) = fs::write(&ref_path, &file_content) {
 169	            tracing::error!("Failed to write offloaded result to {}: {}", ref_path.display(), e);
 170	            // Fallback: keep in context
 171	            return OffloadedResult::new(
 172	                content.to_string(),
 173	                format!("fallback_{}", step_index),
 174	                None,
 175	            );
 176	        }

 178	        tracing::info!(
 179	            "Offloaded tool result: {} -> {} ({} bytes)",
 180	            tool_name,
 181	            ref_path.display(),
 182	            file_content.len()
 183	        );
 184	
 185	        OffloadedResult::new(summary, node_id, Some(ref_path))
 186	    }
